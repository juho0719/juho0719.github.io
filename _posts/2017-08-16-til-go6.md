리플렉션 함수 생성
```
func s(args []reflect.Value) []reflect.Value {  // 매개변수와 리턴 값은 반드시 []reflect.Value를 사용
    c := args[0] + args[1]
    return c
}

var sum func(int a, int b) int {
    f := reflect.ValueOf(&sum).Elem()   // sum의 주소를 넘긴 후 Elem()으로 값을 가져옴
    i := reflect.ValueOf(f.Type(), s)   // s의 함수 정보를 생성
    f.Set(i)    // sum의 값 정보인 f에 s의 함수 정보 i를 설정하여 연결
    sum(1, 2)

}
```
